BEGIN
	DECLARE num[10] AS INTEGER ARRAY 
	PROMPT "enter 10 num"
	FOR INDEX 1 TO 10
		ACCEPT NUM[10]
	END FOR
	PRINT"even num are "
	FOR INDEX 1 TO 10
		IF NUM[INDEX]%2 = 0
		PRINT num[INDEX]
	ENDFOR
	PRINT"odd num are "
	FOR INDEX 1 TO 10
		IF NUM[INDEX]%2 != 0
		PRINT num[INDEX]
	ENDFOR
END
